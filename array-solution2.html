Java 8 Array Stream Problem Solving

1. Find all prime numbers


List primes = Arrays.stream(arr)
  .filter(n -> n > 1 && IntStream.rangeClosed(2, (int)Math.sqrt(n)).allMatch(i -> n % i != 0))
  .boxed().collect(Collectors.toList());

2. Count numbers > average


double avg = Arrays.stream(arr).average().orElse(0);
long greaterThanAvg = Arrays.stream(arr).filter(n -> n > avg).count();

3. Frequency of each element


Map freq = Arrays.stream(arr).boxed()
  .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));

4. Top 3 maximum numbers


List top3 = Arrays.stream(arr).boxed()
  .sorted(Comparator.reverseOrder()).distinct().limit(3).collect(Collectors.toList());

5. Number closest to zero


int closestToZero = Arrays.stream(arr).boxed()
  .min(Comparator.comparingInt(Math::abs)).orElse(0);

6. Product of all numbers


int product = Arrays.stream(arr).reduce(1, (a, b) -> a * b);

7. Convert to CSV


String csv = Arrays.stream(arr).mapToObj(String::valueOf)
  .collect(Collectors.joining(", "));

8. Find duplicates


Set seen = new HashSet<>();
Set duplicates = Arrays.stream(arr)
  .filter(n -> !seen.add(n)).boxed().collect(Collectors.toSet());

10. Partition even/odd


Map> evenOdd = Arrays.stream(arr).boxed()
  .collect(Collectors.partitioningBy(n -> n % 2 == 0));

11. Group by even/odd


Map> evenOddGroup = Arrays.stream(arr).boxed()
  .collect(Collectors.groupingBy(n -> (n % 2 == 0) ? "Even" : "Odd"));

12. Appear exactly once


List once = freq.entrySet().stream()
  .filter(e -> e.getValue() == 1).map(Map.Entry::getKey)
  .collect(Collectors.toList());

13. Remove negatives


int[] nonNegative = Arrays.stream(arr).filter(n -> n >= 0).toArray();

14. Replace negatives with 0


int[] replacedNeg = Arrays.stream(arr).map(n -> n < 0 ? 0 : n).toArray();

15. Sort by absolute value


int[] absSorted = Arrays.stream(arr).boxed()
  .sorted(Comparator.comparingInt(Math::abs))
  .mapToInt(Integer::intValue).toArray();

16. Longest increasing subsequence


List increasing = new ArrayList<>();
Arrays.stream(arr).forEach(n -> {
  if (increasing.isEmpty() || n > increasing.get(increasing.size() - 1))
    increasing.add(n);
});
