14. Find Common Elements in Arrays Using Java 8 Stream API
import java.util.*;
import java.util.stream.*;

public class CommonElementsWithStreams {
    public static void main(String[] args) {
        int[] a = {2, 3, 4, 9};
        int[] b = {2, 3, 7};
        int[] c = {2, 3, 1};

        // Convert arrays to Set for faster lookup
        Set<Integer> setB = Arrays.stream(b).boxed().collect(Collectors.toSet());
        Set<Integer> setC = Arrays.stream(c).boxed().collect(Collectors.toSet());

        // Find common elements using stream
        List<Integer> common = Arrays.stream(a)
                .boxed()
                .filter(setB::contains)
                .filter(setC::contains)
                .collect(Collectors.toList());

        System.out.println("Common elements: " + common);
    }
}

✅ Output:

Common elements: [2, 3]

✅ Explanation:

  • Arrays.stream(a).boxed() – Converts int[] to Stream<Integer>.
  • .filter(setB::contains) – Keeps only elements present in array b.
  • .filter(setC::contains) – Keeps only elements present in array c.
  • collect(Collectors.toList()) – Collects the result into a List<Integer>.
15. Find Duplicates in an Integer Array
import java.util.Arrays; import java.util.HashSet; import java.util.Set; import java.util.stream.Collectors; public class FindDuplicates { public static void main(String[] args) { int[] numbers = {1, 2, 3, 2, 4, 1, 5, 6, 5}; Set<Integer> duplicates = findDuplicateElements(numbers); System.out.println("Duplicate elements: " + duplicates); // Output: [1, 2, 5] } public static Set<Integer> findDuplicateElements(int[] array) { Set<Integer> seen = new HashSet<>(); return Arrays.stream(array) .boxed() .filter(n -> !seen.add(n)) .collect(Collectors.toSet()); } }
16. Find Frequency of Each Element in an Array
import java.util.Arrays; import java.util.Map; import java.util.function.Function; import java.util.stream.Collectors; public class FrequencyCounter { public static void main(String[] args) { int[] numbers = {1, 2, 2, 3, 1, 4, 2}; Map<Integer, Long> frequencyMap = Arrays.stream(numbers) .boxed() .collect(Collectors.groupingBy(Function.identity(), Collectors.counting())); System.out.println(frequencyMap); // Output: {1=2, 2=3, 3=1, 4=1} } }
