Java 8 List Problem Solving

1. Frequency of elements

Map<Integer, Long> freqMap = list.stream()
    .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
// Output: {1=1, 3=3, 5=2, 7=1, 9=1}

2. Remove duplicates

List<Integer> distinctList = list.stream().distinct().collect(Collectors.toList());
// Output: [3, 5, 9, 1, 7]

3. Max and Min

int max = list.stream().max(Integer::compareTo).orElse(-1);
int min = list.stream().min(Integer::compareTo).orElse(-1);
// Output: Max = 9, Min = 1

4. Sum and Average

int sum = list.stream().mapToInt(i -> i).sum();
double avg = list.stream().mapToInt(i -> i).average().orElse(0.0);
// Output: Sum = 36, Avg = 4.5

5. Sort Ascending and Descending

List<Integer> asc = list.stream().sorted().collect(Collectors.toList());
List<Integer> desc = list.stream().sorted(Comparator.reverseOrder()).collect(Collectors.toList());
// Output: Asc: [1, 3, 3, 3, 5, 5, 7, 9]
//         Desc: [9, 7, 5, 5, 3, 3, 3, 1]

6. Count even numbers

long evenCount = list.stream().filter(i -> i % 2 == 0).count();
// Output: 0

7. Square each element

List<Integer> squared = list.stream().map(i -> i * i).collect(Collectors.toList());
// Output: [9, 25, 9, 81, 1, 25, 49, 9]

8. Filter numbers > 5

List<Integer> gtFive = list.stream().filter(i -> i > 5).collect(Collectors.toList());
// Output: [9, 7]

9. Group by Even/Odd

Map<String, List<Integer>> parityGroup = list.stream()
    .collect(Collectors.groupingBy(i -> i % 2 == 0 ? "Even" : "Odd"));
// Output: {Odd=[3, 5, 3, 9, 1, 5, 7, 3]}

10. First element > 5

Optional<Integer> firstGT5 = list.stream().filter(i -> i > 5).findFirst();
// Output: 9

11. Count distinct elements

long distinctCount = list.stream().distinct().count();
// Output: 5

12. Find duplicates

Set<Integer> duplicates = list.stream()
    .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()))
    .entrySet().stream().filter(e -> e.getValue() > 1)
    .map(Map.Entry::getKey).collect(Collectors.toSet());
// Output: [3, 5]

13. Comma-separated string

String joined = list.stream().map(String::valueOf).collect(Collectors.joining(", "));
// Output: "3, 5, 3, 9, 1, 5, 7, 3"

14. All elements positive?

boolean allPositive = list.stream().allMatch(i -> i > 0);
// Output: true

15. Partition into Even and Odd

Map<Boolean, List<Integer>> partitioned = list.stream()
    .collect(Collectors.partitioningBy(i -> i % 2 == 0));
// Output: {false=[3, 5, 3, 9, 1, 5, 7, 3], true=[]}
