1) Separate odd and even numbers 
// Using partitioningBy (gives true/false keys)
Map<Boolean, List<Integer>> result = list.stream()
    .collect(Collectors.partitioningBy(i -> i % 2 == 0));
	
output  :{false=[3, 5, 11, 43, 33, 65], true=[22, 12, 24, 32, 44]}

	

// Using groupingBy to get "Even" and "Odd" keys
Map<String, List<Integer>> resultNamed = list.stream()
    .collect(Collectors.groupingBy(i -> i % 2 == 0 ? "Even" : "Odd"));
	
		
output  :{Even=[22, 12, 24, 32, 44], Odd=[3, 5, 11, 43, 33, 65]}
