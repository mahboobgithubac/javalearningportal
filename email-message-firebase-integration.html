exports.sendEmailAlert = functions.firestore
  .document("orders/{orderId}")
  .onCreate(async (snap, context) => {
    const order = snap.data();

    let transporter = nodemailer.createTransport({
      service: 'gmail',
      auth: {
        user: 'your_email@gmail.com',
        pass: 'your_app_password' // Use App Password
      }
    });

    let info = await transporter.sendMail({
      from: '"Order Alerts" <your_email@gmail.com>',
      to: 'destination_email@gmail.com',
      subject: "ğŸ›’ New Order Received",
      text: `Order: ${order.product} x ${order.quantity}`
    });

    console.log("Email sent: ", info.messageId);
  });
