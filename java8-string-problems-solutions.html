Java 8 String Problem Solving Questions & Solutions

1. Count character frequency
String input1 = "banana"; Map<Character, Long> freq = input1.chars() .mapToObj(c -> (char) c) .collect(Collectors.groupingBy(Function.identity(), Collectors.counting())); System.out.println("Char Frequency: " + freq);
2. First non-repeating character
String input2 = "swiss"; Character firstNonRepeat = input2.chars() .mapToObj(c -> (char) c) .collect(Collectors.groupingBy(Function.identity(), LinkedHashMap::new, Collectors.counting())) .entrySet().stream() .filter(e -> e.getValue() == 1) .map(Map.Entry::getKey) .findFirst().orElse(null); System.out.println("First non-repeating: " + firstNonRepeat);
3. Reverse string
String input3 = "hello"; String reversed = new StringBuilder(input3).reverse().toString(); System.out.println("Reversed: " + reversed);
4. Anagram check
String a1 = "listen", a2 = "silent"; boolean isAnagram = a1.length() == a2.length() && a1.chars().sorted().boxed().collect(Collectors.toList()) .equals(a2.chars().sorted().boxed().collect(Collectors.toList())); System.out.println("Anagram: " + isAnagram);
5. Remove duplicate characters
String input5 = "programming"; String noDupes = input5.chars().mapToObj(c -> (char) c) .distinct() .map(String::valueOf) .collect(Collectors.joining()); System.out.println("Removed duplicates: " + noDupes);
6. Most frequent character
String input6 = "aabbbbcc"; Character mostFreq = input6.chars() .mapToObj(c -> (char) c) .collect(Collectors.groupingBy(Function.identity(), Collectors.counting())) .entrySet().stream() .max(Map.Entry.comparingByValue()) .get().getKey(); System.out.println("Most frequent: " + mostFreq);
7. Sort characters alphabetically
String input7 = "dcba"; String sorted = input7.chars() .sorted() .mapToObj(c -> String.valueOf((char) c)) .collect(Collectors.joining()); System.out.println("Sorted: " + sorted);
8. Count vowels in a string
String input8 = "education"; long vowels = input8.toLowerCase().chars() .filter(c -> "aeiou".indexOf(c) >= 0) .count(); System.out.println("Vowel count: " + vowels);
9. Palindrome check
String input9 = "madam"; boolean isPalin = input9.equals(new StringBuilder(input9).reverse().toString()); System.out.println("Is palindrome: " + isPalin);
10. Convert to Title Case
String input10 = "hello world"; String title = Arrays.stream(input10.split(" ")) .map(w -> w.substring(0, 1).toUpperCase() + w.substring(1)) .collect(Collectors.joining(" ")); System.out.println("Title case: " + title);
11. Duplicate characters in a string
String input11 = "mississippi"; Set<Character> dupes = input11.chars().mapToObj(c -> (char) c) .collect(Collectors.groupingBy(Function.identity(), Collectors.counting())) .entrySet().stream() .filter(e -> e.getValue() > 1) .map(Map.Entry::getKey) .collect(Collectors.toSet()); System.out.println("Duplicates: " + dupes);
12. Group characters by frequency
String input12 = "aaabbc"; Map<Long, List<Character>> groupFreq = input12.chars().mapToObj(c -> (char) c) .collect(Collectors.groupingBy(Function.identity(), Collectors.counting())) .entrySet().stream() .collect(Collectors.groupingBy(Map.Entry::getValue, Collectors.mapping(Map.Entry::getKey, Collectors.toList()))); System.out.println("Grouped by frequency: " + groupFreq);
13. Count words in a string
String input13 = "Java 8 streams are powerful"; long wordCount = Arrays.stream(input13.trim().split("\\\\s+")).count(); System.out.println("Word count: " + wordCount);
14. Extract digits from a string
String input14 = "a1b2c3"; String digits = input14.chars() .filter(Character::isDigit) .mapToObj(c -> String.valueOf((char) c)) .collect(Collectors.joining()); System.out.println("Digits extracted: " + digits);
15. Check if all characters are unique
String input15 = "abcdef"; boolean allUnique = input15.length() == input15.chars().distinct().count(); System.out.println("All unique: " + allUnique);
