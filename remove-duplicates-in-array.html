Remove Duplicates from Array in Java

1. Using Java 8 Streams 
import java.util.Arrays;
import java.util.stream.Collectors;

public class RemoveDuplicates {
    public static void main(String[] args) {
        Integer[] numbers = {1, 2, 2, 3, 4, 4, 5};

        Integer[] unique = Arrays.stream(numbers)
                                 .distinct()
                                 .toArray(Integer[]::new);

        System.out.println(Arrays.toString(unique));
    }
}
2. Using Set (LinkedHashSet) 
import java.util.Arrays;
import java.util.LinkedHashSet;
import java.util.Set;

public class RemoveDuplicates {
    public static void main(String[] args) {
        Integer[] numbers = {1, 2, 2, 3, 4, 4, 5};

        Set<Integer> set = new LinkedHashSet<>(Arrays.asList(numbers));
        Integer[] unique = set.toArray(new Integer[0]);

        System.out.println(Arrays.toString(unique));
    }
}
3. For Primitive int Array 
import java.util.Arrays;
import java.util.stream.IntStream;

public class RemoveDuplicates {
    public static void main(String[] args) {
        int[] nums = {1, 2, 2, 3, 4, 4, 5};

        int[] unique = IntStream.of(nums)
                                .distinct()
                                .toArray();

        System.out.println(Arrays.toString(unique));
    }
}
