✅ Example: Sort HashMap<String, Integer> by Age (Value)

✅ Java 8 Code

import java.util.*;
import java.util.stream.*;

public class SortMapByAge {
    public static void main(String[] args) {
        Map<String, Integer> map = new HashMap<>();
        map.put("Alice", 30);
        map.put("Bob", 25);
        map.put("Charlie", 35);

        System.out.println("Original Map:");
        map.forEach((k, v) -> System.out.println(k + " -> " + v));

        // Sort by value (age)
        LinkedHashMap<String, Integer> sortedMap = map.entrySet()
            .stream()
            .sorted(Map.Entry.comparingByValue())
            .collect(
                Collectors.toMap(
                    Map.Entry::getKey,
                    Map.Entry::getValue,
                    (e1, e2) -> e1,
                    LinkedHashMap::new
                )
            );

        System.out.println("\nSorted Map by Age:");
        sortedMap.forEach((k, v) -> System.out.println(k + " -> " + v));
    }
}

✅ Output

Original Map:
Alice -> 30
Bob -> 25
Charlie -> 35

Sorted Map by Age:
Bob -> 25
Alice -> 30
Charlie -> 35

✅ Descending Order

Change the stream sort line to reverse order:

.sorted(Map.Entry.<String, Integer>comparingByValue().reversed())
