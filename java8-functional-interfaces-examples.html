Functional Interfaces

Java 8 Built-in Functional Interfaces

Predicate<T>

Purpose: Tests a condition and returns a boolean result.

import java.util.function.Predicate;

public class PredicateExample {
    public static void main(String[] args) {
        Predicate<String> isLong = str -> str.length() > 5;
        System.out.println(isLong.test("Mahboob"));
    }
}
Output:
true

Function<T, R>

Purpose: Accepts a value and returns a transformed result.

import java.util.function.Function;

public class FunctionExample {
    public static void main(String[] args) {
        Function<String, Integer> length = str -> str.length();
        System.out.println(length.apply("Mahboob"));
    }
}
Output:
7

Consumer<T>

Purpose: Accepts a value and performs an action without returning anything.

import java.util.function.Consumer;

public class ConsumerExample {
    public static void main(String[] args) {
        Consumer<String> greet = name -> System.out.println("Hello " + name);
        greet.accept("Mahboob");
    }
}
Output:
Hello Mahboob

Supplier<T>

Purpose: Supplies a value without accepting input.

import java.util.function.Supplier;

public class SupplierExample {
    public static void main(String[] args) {
        Supplier<String> nameSupplier = () -> "Mahboob";
        System.out.println(nameSupplier.get());
    }
}
Output:
Mahboob

UnaryOperator<T>

Purpose: Accepts and returns a value of the same type.

import java.util.function.UnaryOperator;

public class UnaryOperatorExample {
    public static void main(String[] args) {
        UnaryOperator<String> toUpper = str -> str.toUpperCase();
        System.out.println(toUpper.apply("mahboob"));
    }
}
Output:
MAHBOOB

BinaryOperator<T>

Purpose: Combines two values of the same type and returns one of the same type.

import java.util.function.BinaryOperator;

public class BinaryOperatorExample {
    public static void main(String[] args) {
        BinaryOperator<Integer> add = (a, b) -> a + b;
        System.out.println(add.apply(10, 5));
    }
}
Output:
15

BiFunction<T, U, R>

Purpose: Accepts two values and returns a result.

import java.util.function.BiFunction;

public class BiFunctionExample {
    public static void main(String[] args) {
        BiFunction<String, String, String> concat = (a, b) -> a + " " + b;
        System.out.println(concat.apply("Hello", "World"));
    }
}
Output:
Hello World

BiPredicate<T, U>

Purpose: Tests a condition using two inputs.

import java.util.function.BiPredicate;

public class BiPredicateExample {
    public static void main(String[] args) {
        BiPredicate<String, Integer> isLongerThan = (str, len) -> str.length() > len;
        System.out.println(isLongerThan.test("Mahboob", 5));
    }
}
Output:
true

BiConsumer<T, U>

Purpose: Performs an action using two inputs.

import java.util.function.BiConsumer;

public class BiConsumerExample {
    public static void main(String[] args) {
        BiConsumer<String, Integer> printInfo = (name, age) ->
            System.out.println(name + " is " + age + " years old");
        printInfo.accept("Mahboob", 30);
    }
}
Output:
Mahboob is 30 years old
